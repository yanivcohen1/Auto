	function getTimeDifference(referenceDate, now) {
		var millisecRefTime = referenceDate.getTime();
		var millisecNowTime = now.getTime();
		return (millisecRefTime - millisecNowTime);
	}
	
	function pad(intValue) {
		return (intValue &lt; 10) ? "0" + intValue : intValue;
	}
	
	function formatTimeDiff(intDiff) {
		// if the difference is negative, show 00:00:00
		if (intDiff &gt; 0 ) {
			intDiff = (intDiff/1000); //intDiff is in milliseconds
			hours = calculateUnit((intDiff),3600);
			mins = calculateUnit((intDiff-(hours*3600)),60);
			secs = calculateUnit((intDiff-(hours*3600)-(mins*60)),1);
			return pad(hours) + ":" + pad(mins) + ":" + pad(secs);
		}
		return "00:00:00";
	}
	
	function calculateUnit(secDiff, unitSeconds){
		var tmp = Math.abs((tmp = secDiff/unitSeconds)) &lt; 1? 0 : tmp;
		return Math.abs(tmp &lt; 0 ? Math.ceil(tmp) : Math.floor(tmp));
	}
