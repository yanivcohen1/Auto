	// Google Analytics tracking helper function to calculate how long a user has spent completing a form
	function calculateIntervalSeconds(form_name) {
		var end_date = new Date();
		var end_time = end_date.getTime();
		if (form_name == "Reg") {
			var start_time = document.registrationStartTime;
		} else {
			var start_time = document.depositStartTime;
		}

		if (start_time == null || start_time == "") {
			return "N/A";
		} else {
			var milliseconds_elapsed = end_time - start_time;
			var seconds_elapsed = parseInt(milliseconds_elapsed / 1000);
			return seconds_elapsed;
		}
	}
